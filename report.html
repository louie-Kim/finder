requestAnimationFrame throttling

const scheduleRefresh = useCallback(() => {
  if (rafRef.current !== null) {
    cancelAnimationFrame(rafRef.current);
  }
  rafRef.current = requestAnimationFrame(() => {
    rafRef.current = null;
    refreshMatches();
  });
}, [refreshMatches]);
